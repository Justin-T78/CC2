import java.util.Scanner;

public class FindSquareRootExample1 {
    public static void main(String[] args) {
        System.out.print("Enter a number: ");
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        sc.close();

        int result = squareRoot(n);
        System.out.println("The square root of " + n + " is: " + result);
    }

    // Method to find integer square root rounded down
    public static int squareRoot(int num) {
        if (num < 2) return num;  // sqrt(0) = 0 and sqrt(1) = 1

        int start = 1, end = num / 2, ans = 0;
        while (start <= end) {
            int mid = start + (end - start) / 2;
            long square = (long) mid * mid;

            if (square == num) {
                return mid;
            } else if (square < num) {
                ans = mid;   // store the floor value
                start = mid + 1;
            } else {
                end = mid - 1;
            }
        }
        return ans;
    }
}
